git status - просмотр в каком состоянии находятся файлы

git add <fileName>  - добавление файла в индех (git будет его отслеживать)

git commit -m "Init commit"  - созданин коммита

git log     - история коммитов
git log -p  - выведет историю коммитов с более подробным выводом (описанием)

git show <передаваемый параметр> - показывает объект из параметра (например хэш коммита) в подробностях

git restore <fileName>  - откатывает последние изменения в файле на состояние последнего коммита

git diff            - показывает все изменения после последнего коммита
git diff --staged   - покажет все изменения, которые уже находятся в индексе

ls -la        - показывает список всех папок (в т.ч. и скрытых)
